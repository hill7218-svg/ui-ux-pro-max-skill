## 💾 APP 核心邏輯（程式簡化）

```javascript
// v8.3 核心：掃QR → 拍照 → 存檔

function scanQRCode() {
  // 1. 掃貼紙 QR Code
  const qrData = camera.scanQR();
  // → {sourceId, sku, destLocation, ...}
  
  // 2. 查表帶出入庫單資訊
  const order = ORDERS[qrData.sourceId];
  displayOrderInfo(order);
  
  // 3. 員工A 點擊「視覺確認」checkbox
  // 4. 員工B 按「拍照」
  
  // 5. 自動拍照
  const photo = camera.takePicture();
  
  // 6. 照片上傳 Google Drive
  uploadToGoogleDrive({
    fileName: `plate-${order.boardNum}.jpg`,
    photo: photo,
    metadata: {
      orderId: order.id,
      sku: order.sku,
      timestamp: new Date()
    }
  });
  
  // 7. 記錄到 Google Sheet
  appendToSheet({
    orderId: order.id,
    photoFile: `plate-${order.boardNum}.jpg`,
    status: "✅"
  });
  
  // 8. 更新進度 + 下一板
  updateProgress();
}
```
